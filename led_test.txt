REM 無限ループテスト: IF / GOTO / GOSUB / RETURN の動作確認用
UseLED(1)
Mode(KeyMouse)
LABEL LOOP
SetLED(255,0,0)
WAIT 0.5
SET r = Rand(0,2)
IF r == 0 GOTO KEY_SECTION
IF r == 1 GOTO MOUSE_SECTION
GOTO AFTER

LABEL KEY_SECTION
SetLED(0,255,0)
WAIT 0.2
KeyPress("A")
WAIT 0.1
KeyRelease("A")
GOTO AFTER

LABEL MOUSE_SECTION
SetLED(0,0,255)
MouseMove(20,0,1)
MousePushFor(LEFT, 0.2)
WAIT 0.1
GOTO AFTER


LABEL AFTER
SetLED(0,0,0)
WAIT 0.3
GOTO LOOP
